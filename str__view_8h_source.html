29/* SV_ATTRIB_PURE has no side effects when given the same arguments with the
30 same data, producing the same return value. A SV_Str_view points to char
31 const * data which may change in between SV_Str_view calls. SV_ATTRIB_CONST
32 applies only where no pointers are accessed or dereferenced. Other
33 attributes provide more string safety and opportunities to optimize.
34 Credit Harith on Code Review Stack Exchange. */

36# ifdef __has_attribute
37# if __has_attribute(pure)
40# define SV_ATTRIB_PURE __attribute__((pure))
41# else
44# define SV_ATTRIB_PURE
45# endif
46# if __has_attribute(pure)
49# define SV_ATTRIB_CONST __attribute__((const))
50# else
53# define SV_ATTRIB_CONST
54# endif
55# if __has_attribute(null_terminated_string_arg)
57# define SV_ATTRIB_NULLTERM(...) \
58 __attribute__((null_terminated_string_arg(__VA_ARGS__)))
59# else
61# define SV_ATTRIB_NULLTERM(...)
62# endif
63# else
66# define SV_ATTRIB_PURE
69# define SV_ATTRIB_CONST
71# define SV_ATTRIB_NULLTERM(...)
72# endif
77# define SV_STR_LITERAL(str_literal) "" str_literal ""
78#else
81# define SV_ATTRIB_PURE
84# define SV_ATTRIB_CONST
